\begin{frame}
\frametitle{Motivation}

``Should the guidelines' traditional distinctions between horizontal and vertical mergers be revisited in light of recent economic trends in the modern economy? What aspects of modern market realities may be lost by focusing on these relationships categorically?'' \\
\scriptsize{(\textit{Request for Information on Merger Enforcement}, Question 1g, DOJ and FTC, 2022)}

\bigbreak \begin{itemize}
	\item Few producers operate at only one level of a supply chain
	\begin{itemize}
			\item Mergers frequently combine firms and impact rivals that are already vertically integrated
			\item Nevertheless, standard merger analysis often simplifies a case into purely horizontal and vertical categories
		\end{itemize}
	\item What happens when we incorporate some of these vertical relationships into a typical merger simulation model?
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Research Goals}

\begin{enumerate}
\item Incorporate additional vertical integration into the Sheu and Taragin (2021) merger simulation model
	\begin{itemize}
		\item Allow some merging parties to already be integrated
		\item Allow some rivals to be integrated 
		\item Calculate welfare effects balancing UPP, EDM, RRC
	\end{itemize}
\item Study welfare effects of four types of mergers
	\begin{enumerate}
		\item Vertical: merger between unintegrated upstream and unintegrated downstream firms
		\item Upstream: merger between an unintegrated upstream firm and an upstream firm that may already be integrated with a downstream firm 
		\item Downstream: merger between an unintegrated downstream firm and a downstream firm that may already be integrated with a upstream firm 
		\item Integrated: merger between two vertically integrated firms 
	\end{enumerate}
\item Apply model to Republic/Santek merger (2021) in solid waste management
\end{enumerate}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Findings}

\begin{itemize}
	\item Vertical effects in ``horizontal'' mergers (both upstream and downstream) create net consumer benefits only in limited cases
		\begin{itemize}
			\item Mostly when upstream firms have more bargaining power (when EDM is expected to be larger)
		\end{itemize}
	\item Mergers between two integrated firms are typically net harmful to consumers 
		\begin{itemize}
			\item Only beneficial when upstream firms have a significant edge in bargaining power
		\end{itemize}
	\item The presence of integrated rivals does not have a large impact on realized harm 
		\begin{itemize}
			\item Can somewhat shift outcomes towards less harm, but differences not dramatic
		\end{itemize}
	\item We can model Republic/Santek using limited inputs and generate realistic RRC and EDM effects
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Model Overview}

\begin{itemize}
  \item Each wholesaler $w$ in the set $\mathbb{W}$ offers a single product to retailers $r$ in the set $\mathbb{R}$,
  who in turn sell to consumers
  \begin{enumerate}
    \item Upstream: Nash bargaining between wholesalers and retailers
    \item Downstream: Bertrand differentiated products logit competition between retailers
  \end{enumerate}
	\item All prices (upstream and downstream) assumed to be determined simultaneously
	\item Calibration follows Sheu and Taragin (2021)
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Downstream Bertrand Logit}

\begin{itemize}
  \item Assume firms set prices in Bertrand Nash equilibrium
	\item Market share of wholesaler $w$'s product purchased through retailer $r$ 
  \begin{equation*}
  s_{rw} = \frac{\exp(\delta_{rw} - \alpha p_{rw})}{1 + \sum_{t \in \mathbb{R}} \sum_{x \in \mathbb{W}^t} \exp(\delta_{tx}   - \alpha p_{tx})}
  \end{equation*}
  \item Profit function for retailer $r$ with wholesale fee $p^W_{rw}$ and other costs $c^R_{rw}$
  \begin{equation*}
  \pi^r = \sum_{w \in \mathbb{W}^r} [p_{rw} - p^W_{rw} - c^R_{rw}] s_{rw} M
  \end{equation*}
  \item First order conditions for downstream equilibrium prices
  \begin{equation*}
  \sum_{x \in \mathbb{W}^r} [p_{rx} - p^W_{rx} - c^R_{rx}] \frac{\partial s_{rx}}{\partial p_{rw}} + s_{rw} = 0
  \end{equation*}
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Upstream Nash Bargaining}

\begin{itemize}
  \item Assume a linear wholesale price set by Nash Bargaining with $\lambda$ bargaining power parameter
  \begin{equation*}
  \max_{p^W_{rw}} (\pi^r - d^r (\mathbb{W}^r \setminus \{w\}))^{\lambda} (\pi^w - d^w (\mathbb{R}^w \setminus \{r\}
  ))^{1 - \lambda}
  \end{equation*}
  \item Profit function for wholesaler $w$ with marginal cost $c^W_{rw}$ 
  \begin{equation*}
  \pi^w = \sum_{r \in \mathbb{R}^w} [p^W_{rw} - c^W_{rw}] s_{rw} M
  \end{equation*}
	\item Profit for wholesaler $w$ if it does not sell to retailer $r$ 
  \begin{equation*}
  d^w (\mathbb{R}^w \setminus \{r\}) = \sum_{t \in \mathbb{R}^w \setminus \{r\}} [p^W_{tw} - c^W_{tw}] s_{tw}(\mathbb{W}^r   \setminus \{w\}) M
  \end{equation*}
  \item Profit for retailer $r$ if it does not buy from wholesaler $w$ 
  \begin{equation*}
  d^r (\mathbb{W}^r \setminus \{w\}) = \sum_{x \in \mathbb{W}^r \setminus \{w\}} [p_{rx} - p^W_{rx} - c^R_{rx}] s_{rx}
  (\mathbb{W}^r \setminus \{w\}) M
  \end{equation*}
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Bargaining Equilibrium}

\begin{itemize}
	\item Given assumption of simultaneous pricing upstream and downstream, first order conditions have the form
	\begin{equation*}
	\begin{split}
	&\overbrace{[p^W_{rw} - c^W_{rw}]s_{rw} - \sum_{t \in \mathbb{R}^w \setminus \{r\}} [p^W_{tw} - c^W_{tw}] \Delta s_{tw}(\mathbb{W}^r \setminus \{w\})}^{\text{wholesaler GFT}} = \\
	&\frac{1-\lambda}{\lambda} \left(\underbrace{[p_{rw} - p^W_{rw} - c^R_{rw}]s_{rw} - \sum_{x \in \mathbb{W}^r \setminus \{w\}} [p_{rx} - p^W_{rx} - c^R_{rx}] \Delta s_{rx}(\mathbb{W}^r \setminus \{w\})}_{\text{retailer GFT}}\right)
	\end{split}
	\end{equation*}
  \item A larger disagreement payoff increases bargaining leverage
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Integrated Firm Pricing: Retail}
Assume that retailer $r$ merges with wholesaler $w$. Then $r$'s price downstream for product $w$ is determined by

\begin{equation*}
\begin{split}
\sum_{x \in \mathbb{W}^r \setminus \{w\}} [p_{rx} - p^W_{rx} - c^R_{rx}]\frac{\partial s_{rx}}{\partial p_{rw}} &+ s_{rw}\\
\underbrace{[p_{rw}-c^R_{rw}-c^W_{rw}]\frac{\partial s_{rw}}{\partial p_{rw}}}_\text{EDM effect} &+ \underbrace{\sum_{t \in \mathbb{R}^w \setminus \{r\}} [p^W_{tw} - c^W_{tw}] \frac{\partial s_{tw}}{\partial p_{rw}}}_\text{upstream UPP effect} = 0.
\end{split}
\end{equation*}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Integrated Firm Pricing: Wholesale Price to Rival Retailer}
When wholesaler $w$ bargains with unaffiliated retailer $s$, the wholesale price is determined by

\tiny
\begin{equation*}
\begin{split}
&[p^W_{sw} - c^W_{sw}]s_{sw} - \sum_{t \in \mathbb{R}^w \setminus \{r,s\}} [p^W_{tw} - c^W_{tw}] \Delta s_{tw}(\mathbb{W}^s \setminus \{w\})\\
& - \overbrace{\overbrace{[p_{rw}-c^W_{rw}-c^R_{rw}] \Delta s_{rw}(\mathbb{W}^s \setminus \{w\})}^{\text{indirect EDM effect}} - \sum_{x \in \mathbb{W}^r \setminus \{w\}} [p_{rx} - p^W_{rx} - c^R_{rx}] \Delta s_{rx}(\mathbb{W}^s \setminus \{w\})}^{\text{RRC effect}} =\\
&\frac{1-\lambda}{\lambda} \left([p_{sw} - p^W_{sw} - c^R_{sw}]s_{sw} - \sum_{x \in \mathbb{W}^s \setminus \{w\}} [p_{sx} - p^W_{sx} - c^R_{sx}] \Delta s_{sx}(\mathbb{W}^s \setminus \{w\})\right)
\end{split}
\end{equation*}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
\frametitle{Integrated Firm Pricing: Wholesale Price from Rival Wholesaler}
When retailer $r$ bargains with unaffiliated wholesaler $v$, the wholesale price is determined by

\tiny
\begin{equation*}
\begin{split}
&[p^W_{rv} - c^W_{rv}]s_{rv} - \sum_{t \in \mathbb{R}^v \setminus \{r\}} [p^W_{tv} - c^W_{tv}] \Delta s_{tv}(\mathbb{W}^r \setminus \{v\}) = \\
&\frac{1-\lambda}{\lambda} \left([p_{rv} - p^W_{rv} - c^R_{rv}]s_{rv} - \sum_{x \in \mathbb{W}^r \setminus \{w,v\}} [p_{rx} - p^W_{rx} - c^R_{rx}] \Delta s_{rx}(\mathbb{W}^r \setminus \{v\})\right.\\
&\left.- \underbrace{[p_{rw} - c^W_{rw} - c^R_{rw}] \Delta s_{rw}(\mathbb{W}^r \setminus \{v\})}_{\text{EDM recapture effect}}-\underbrace{\sum_{t \in \mathbb{R}^w \setminus \{r\}} [p^W_{tw} - c^W_{tw}] \Delta s_{tw}(\mathbb{W}^r \setminus \{v\})}_{\text{wholesale recapture leverage effect}}\right)
\end{split}
\end{equation*}

\end{frame}

\begin{frame}
\frametitle{Data Generating Process}

\begin{itemize}
\item Simulate markets with 2, 3, 4, or 5 wholesalers/retailers
\item Have 0 to 4 non-siloed integrated incumbents pre-merger
	\begin{itemize}
		\item 2 to 6 for integrated mergers
	\end{itemize}
\item Bargaining power parameter ranges from 0.1 to 0.9 in 0.1 increments
\item Eliminate some instances
	\begin{itemize}
		\item Unprofitable mergers
		\item Downstream markets that do not pass the Hypothetical Monopolist Test
	\end{itemize}
\item Results in approximately 1.6 million markets
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Summary}

\begin{itemize}
\item We study the impacts of additional vertical integration on the welfare effects of mergers 
	\begin{itemize}
		\item Allow merging parties to already be integrated
		\item Allow rivals to be integrated
	\end{itemize}
\item Most merger simulations show net negative impacts on consumers
	\begin{itemize}
		\item Typically only result in net gains when upstream bargaining power is high (more potential for EDM)
		\item Presence of integrated rivals does not strongly protect consumers from harm 
	\end{itemize}
\item Model shows the effects of EDM and RRC in Republic/Santek
\item Appendix to paper extends to case of increasing marginal costs
	\begin{itemize}
		\item Allow marginal costs to increase linearly
		\item Merger simulations tend to show more harm than under constant marginal costs
	\end{itemize}
\end{itemize}
